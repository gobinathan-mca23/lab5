inventory = {}

def add_item():

    item = input("Enter item name: ").strip().lower()

    qty = int(input("Enter quantity: "))

    inventory[item] = inventory.get(item, 0) + qty

    print(f"{qty} {item}(s) added.")

def update_item():

    item = input("Enter item name to update: ").strip().lower()

    if item in inventory:

        inventory[item] = int(input("Enter new quantity: "))

        print(f"{item} quantity updated.")

    else:

        print("Item not found!")

def remove_item():

    item = input("Enter item name to remove: ").strip().lower()

    if inventory.pop(item, None) is not None:

        print(f"{item} removed.")

    else:

        print("Item not found!")


def view_inventory():

    if not inventory:

        print("Inventory is empty.")

    else:

        print("Inventory List:")

        for item, qty in inventory.items():

            print(f"{item}: {qty}")


def main():

    while True:

        print("\n--- Inventory Management ---")

        print("1. Add Item\n2. Update Item Quantity\n3. Remove Item\n4. View Inventory\n5. Exit")

        choice = input("Choose an option: ")

        if choice == '1':

            add_item()

        elif choice == '2':

            update_item()

        elif choice == '3':

            remove_item()

        elif choice == '4':

            view_inventory()

        elif choice == '5':

            print("Exiting program.")

            break

        else:

            print("Invalid choice, try again.")


if __name__ == "__main__":

    main()
